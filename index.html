def extract_exception_details( self, accp_path, filtered_path, investigate_sheet="Investigate", source_sheet="Sheet1" ): try: import xlwings as xw import pandas as pd import re app = xw.App(visible=True) wb_accp = app.books.open(accp_path) wb_filtered = app.books.open(filtered_path) ws_inv = wb_accp.sheets[investigate_sheet] ws_source = wb_filtered.sheets[source_sheet] # ========================= # 🔹 LOAD INVESTIGATE (A–C) # ========================= last_row = ws_inv.range("A" + str(ws_inv.cells.last_cell.row)).end("up").row df_inv = ws_inv.range(f"A1:C{last_row}").options( pd.DataFrame, header=1, index=False ).value df_inv.columns = [str(c).strip() for c in df_inv.columns] if df_inv.dropna(how="all").empty: print("ℹ️ No investigate data — skipping") wb_filtered.close() wb_accp.save() return acc_l, serial_l, amt_l = df_inv.columns[:3] # ========================= # 🔹 LOAD SOURCE (FULL SHEET) # ========================= raw = ws_source.used_range.value df_raw = pd.DataFrame(raw) # ========================= # 🔍 FIND HEADER ROW # ========================= header_row_idx = None for i in range(len(df_raw)): row_str = " ".join([str(x) for x in df_raw.iloc[i].values]) if "bank" in row_str.lower(): header_row_idx = i break if header_row_idx is None: raise Exception("❌ Header row not found") # ========================= # 🔹 BUILD HEADER # ========================= raw_header = df_raw.iloc[header_row_idx].values header = [] for i, val in enumerate(raw_header): val = str(val).strip() if val == "" or val.lower() == "nan": header.append(f"COL_{i}") else: header.append(val) # ========================= # 🔹 BUILD DATAFRAME # ========================= df_source = df_raw.iloc[header_row_idx + 1:].reset_index(drop=True) df_source = df_source.iloc[:, :len(header)] df_source.columns = header # Remove junk rows containing "process" df_source = df_source[ ~df_source.astype(str).apply( lambda r: r.str.contains("process", case=False, na=False).any(), axis=1 ) ].reset_index(drop=True) # ========================= # 🔍 FIND REQUIRED COLUMNS # ========================= def find_col(cols, keyword): for c in cols: if keyword in str(c).lower(): return c return None acc_col = find_col(header, "account") or find_col(header, "bank") serial_col = find_col(header, "serial") amt_col = find_col(header, "amount") if not acc_col or not serial_col or not amt_col: raise Exception(f"❌ Column detection failed: {header}") print(f"📌 Detected columns → Account: '{acc_col}' | Serial: '{serial_col}' | Amount: '{amt_col}'") # ========================= # 🔥 EXTRACT VALUES BY PATTERN (row-wide scan) # FIX: This correctly finds Account/Serial/Amount regardless of column shift. # ========================= def extract_values(row): acc = "" serial = "" amount = "" for val in row: if pd.isna(val): continue s = str(val).replace(",", "").strip() # Account: long digit string (≥6 digits) if re.fullmatch(r'\d{6,}', s): acc = s # Serial: short digit string (1–3 digits) — only grab first match elif re.fullmatch(r'\d{1,3}', s) and serial == "": serial = s # Amount: numeric value > 1000 elif re.match(r'^\d+(\.\d+)?$', s) and float(s) > 1000: amount = s return pd.Series([acc, serial, amount]) df_source[["ACC_VAL", "SER_VAL", "AMT_VAL"]] = df_source.apply(extract_values, axis=1) print(f"🔍 Sample extracted values:\n{df_source[['ACC_VAL','SER_VAL','AMT_VAL']].head(5)}") # ========================= # 🔹 NORMALIZE FOR MATCHING # ========================= def clean_num(x): try: return str(int(float(str(x).replace(",", "")))) except: return "" df_inv[serial_l] = df_inv[serial_l].astype(str).str.replace(",", "").str.strip() df_inv[amt_l] = pd.to_numeric(df_inv[amt_l], errors="coerce") # ========================= # 🔹 MATCH + EXTRACT # Primary match: Amount + Serial # Fallback match: Amount only # Workaround: Account & Serial stamped from Investigate row (source columns are unreliable) # ========================= results = [] acc_pos = header.index(acc_col) if acc_col in header else None serial_pos = header.index(serial_col) if serial_col in header else None amt_pos = header.index(amt_col) if amt_col in header else None for _, row in df_inv.iterrows(): if pd.isna(row[amt_l]): continue inv_acc = str(row[acc_l]).strip() inv_ser = clean_num(row[serial_l]) inv_amt = clean_num(row[amt_l]) # Primary: match on both amount and serial matches = df_source[ (df_source["AMT_VAL"].apply(clean_num) == inv_amt) & (df_source["SER_VAL"].apply(clean_num) == inv_ser) ] # Fallback: match on amount only if matches.empty: print(f"⚠️ No amt+serial match for amt={inv_amt}, serial={inv_ser} — trying amount-only fallback") matches = df_source[ df_source["AMT_VAL"].apply(clean_num) == inv_amt ] if matches.empty: print(f"❌ No match found for amt={inv_amt}, serial={inv_ser}") continue for idx in matches.index: row_data = df_source.iloc[idx].tolist()[:10] # ✅ WORKAROUND: Stamp Account & Serial directly from Investigate sheet # (source extraction is unreliable — Investigate values are the ground truth) if acc_pos is not None and acc_pos < len(row_data): row_data[acc_pos] = inv_acc if serial_pos is not None and serial_pos < len(row_data): row_data[serial_pos] = inv_ser if amt_pos is not None and amt_pos < len(row_data): row_data[amt_pos] = inv_amt results.append(row_data) # Companion row below — stamp same Account & Serial (same transaction group) if idx + 1 < len(df_source): row_below = df_source.iloc[idx + 1].tolist()[:10] if acc_pos is not None and acc_pos < len(row_below): row_below[acc_pos] = inv_acc if serial_pos is not None and serial_pos < len(row_below): row_below[serial_pos] = inv_ser if amt_pos is not None and amt_pos < len(row_below): row_below[amt_pos] = inv_amt results.append(row_below) print(f"✅ Total result rows collected: {len(results)}") # ========================= # 🔹 CLEAN None / NaN → "" # ========================= clean_results = [ ["" if (isinstance(v, float) and pd.isna(v)) or str(v).strip().lower() in ("none", "nan") else v for v in row] for row in results ] # ========================= # 🔹 FILTER OUT PLACEHOLDER COLUMNS # ========================= valid_indices = [ i for i, col in enumerate(header[:10]) if str(col).strip().lower() not in ["none", "nan"] and not str(col).startswith("COL_") ] final_header = [header[i] for i in valid_indices] final_results = [[row[i] for i in valid_indices if i < len(row)] for row in clean_results] # ========================= # 🔹 WRITE TO EXCEPTION DETAILS # ========================= if final_results: # Remove sheet if it already exists (re-run safety) existing = [s.name for s in wb_accp.sheets] if "Exception Details" in existing: wb_accp.sheets["Exception Details"].delete() ws_exc = wb_accp.sheets.add("Exception Details") ws_exc.range("A1").value = final_header ws_exc.range("A2").value = final_results print(f"📄 'Exception Details' sheet written with {len(final_results)} rows.") else: print("⚠️ No results to write — 'Exception Details' sheet NOT created.") print(" Check that Investigate amounts/serials actually exist in the source sheet.") wb_accp.save() wb_filtered.close() print("✅ Exception extraction complete") except Exception as e: import traceback print(f"❌ Error: {e}") traceback.print_exc() finally: app.quit()
